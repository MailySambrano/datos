#!/bin/bash

# Variables
carpeta_origen="/ruta/a/tu/carpeta_origen"  # Ruta de la carpeta en Ubuntu
carpeta_destino="/mnt/windows"  # Punto de montaje SMB en Ubuntu
usuario="tu_usuario"  # Usuario de Windows
contrasena="tu_contrasena"  # Contraseña de Windows

# Montar la carpeta compartida de Windows (si no está montada)
if ! mount | grep -q "$carpeta_destino"; then
  sudo mount -t cifs //192.168.1.100/compartido "$carpeta_destino" -o username=$usuario,password=$contrasena
fi

# Copiar los archivos
cp -r "$carpeta_origen"/* "$carpeta_destino"

# Desmontar la carpeta compartida (opcional)
sudo umount "$carpeta_destino"
